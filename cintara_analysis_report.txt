            const input = document.getElementById('chat-input');

            const message = input.value.trim();

            if (!message) return;

            responseDiv.style.display = 'block';
            responseDiv.textContent = 'Thinking...';

                const response = await fetch('/api/chat', {

                    body: JSON.stringify({ message: message })

                if (!response.ok) {
                    throw new Error(`HTTP ${response.status}: ${response.statusText}`);

        async function exportTaxBit() {
            const input = document.getElementById('taxbit-address');
            const responseDiv = document.getElementById('taxbit-response');
            const address = input.value.trim();

            if (!address) {
                alert('Please enter a Cintara address');
                return;
            }

            // Validate address format (accept both Cosmos bech32 and Ethereum hex formats)
            const isValidAddress = address.startsWith('cintara1') ||
                                 address.startsWith('cosmos1') ||
                                 (address.startsWith('0x') && address.length === 42);

            if (!isValidAddress) {
                alert('Please enter a valid address:\n- Cintara format: cintara1...\n- Cosmos format: cosmos1...\n- Ethereum format: 0x...');
                return;
            }

            responseDiv.style.display = 'block';
            responseDiv.textContent = 'Generating CSV export...';

            try {
                const response = await fetch(`/api/taxbit/export/${encodeURIComponent(address)}`);

                if (response.ok) {
                    // Create a download link
                    const blob = await response.blob();
                    const url = window.URL.createObjectURL(blob);
                    const a = document.createElement('a');
                    a.href = url;
                    a.download = `cintara_taxbit_export_${address.substring(0, 12)}_${new Date().toISOString().slice(0, 10)}.csv`;
                    document.body.appendChild(a);
                    a.click();
                    window.URL.revokeObjectURL(url);
                    document.body.removeChild(a);

                    responseDiv.innerHTML = `<div style="color: #2da44e;"><strong>✅ CSV Export Successful!</strong></div>` +
                        `<div>Your TaxBit-compatible CSV has been downloaded.</div>` +
                        `<div style="margin-top: 0.5rem; font-size: 0.875rem; color: #656d76;">You can now import this file into TaxBit for tax reporting.</div>`;
                } else {
                    const errorData = await response.json();
                    responseDiv.textContent = `Error: ${errorData.detail || 'Failed to export CSV'}`;
                }
